CREATE  TABLE IF NOT EXISTS `CONTRACT` (
  `LOANID` CHAR(12) NOT NULL ,
  `CONTRACTSEQ` INT NOT NULL ,
  `LOGNO` BIGINT NOT NULL ,
  `CREATEDBY` INT NOT NULL ,
  `CREATEDDT` TIMESTAMP NOT NULL ,
  PRIMARY KEY (`LOANID`, `CONTRACTSEQ`) ,
  INDEX `CONT_LOANID` (`LOANID` ASC) ,
  CONSTRAINT `CONT_LOANID`
    FOREIGN KEY (`LOANID` )
    REFERENCES `LOAN` (`LOANID` )
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB

CREATE  TABLE IF NOT EXISTS `LOAN` (
  `LOANID` CHAR(12) NOT NULL ,
  `CUSTID` CHAR(12) NOT NULL ,
  `LOGNO` BIGINT NOT NULL ,
  `CREATEDBY` INT NOT NULL ,
  `CREATEDDT` TIMESTAMP NOT NULL ,
  PRIMARY KEY (`LOANID`, `CUSTID`) ,
  INDEX `LOAN_CUSTID` (`CUSTID` ASC) ,
  CONSTRAINT `LOAN_CUSTID`
    FOREIGN KEY (`CUSTID` )
    REFERENCES `CUSTOMER` (`CUSTID` )
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB

CREATE  TABLE IF NOT EXISTS `CUSTOMER` (
  `CUSTID` CHAR(12) NOT NULL ,
  `PERSONID` BIGINT NOT NULL ,
  `LOGNO` BIGINT NOT NULL ,
  `CREATEDBY` INT NOT NULL ,
  `CREATEDDT` TIMESTAMP NOT NULL ,
  PRIMARY KEY (`CUSTID`, `PERSONID`) ,
  INDEX `CUST_PERSONID` (`PERSONID` ASC) ,
  CONSTRAINT `CUST_PERSONID`
    FOREIGN KEY (`PERSONID` )
    REFERENCES `PERSON` (`PERSONID` )
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB

CREATE  TABLE IF NOT EXISTS `PERSON` (
  `PERSONID` BIGINT NOT NULL AUTO_INCREMENT ,
  `LOGNO` BIGINT NOT NULL ,
  `CREATEDBY` INT NOT NULL ,
  `CREATEDDT` TIMESTAMP NOT NULL ,
  PRIMARY KEY (`PERSONID`) )
ENGINE = InnoDB
